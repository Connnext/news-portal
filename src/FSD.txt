src/
 ├── app/                # Инициализация приложения, провайдеры, настройки
 ├── shared/             # Переиспользуемые компоненты, утилиты, API-интерфейсы
 ├── entities/           # Базовые сущности (посты, пользователи, комментарии)
 ├── features/           # Отдельные функции (лайки, подписки, фильтры, авторизация)
 ├── widgets/            # Крупные UI-виджеты (лента новостей, профиль, меню навигации)
 ├── pages/              # Страницы (Лента, Профиль, Авторизация)
 ├── processes/          # Глобальные процессы (например, аутентификация)
 ├── index.tsx           # Точка входа в приложение
